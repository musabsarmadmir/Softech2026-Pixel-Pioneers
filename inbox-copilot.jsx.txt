      {step === 3 && (
        <div style={{ minHeight: "85vh", display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center", padding: 40 }}>
          <div style={{ width: 64, height: 64, borderRadius: "50%", border: "3px solid #1e2a4a", borderTopColor: "#3b82f6", animation: "spin 1s linear infinite", marginBottom: 32 }} />
          <style>{`@keyframes spin{to{transform:rotate(360deg)}}`}</style>
          <h3 className="syne" style={{ fontSize: 22, fontWeight: 700, marginBottom: 16 }}>Analyzing Your Inbox</h3>
          <div className="card scan-line" style={{ padding: "14px 24px", minWidth: 320, textAlign: "center" }}>
            <span className="mono" style={{ fontSize: 14, color: "#3b82f6" }}>{PROC_STEPS[procIdx]}</span>
          </div>
          <div style={{ display: "flex", gap: 6, marginTop: 24 }}>
            {PROC_STEPS.map((_, i) => (
              <div key={i} style={{ width: i === procIdx ? 20 : 6, height: 6, borderRadius: 3, background: i <= procIdx ? "#3b82f6" : "#1e2a4a", transition: "all .3s" }} />
            ))}
          </div>
        </div>
      )}

      {/* STEP 4 — RESULTS */}
      {step === 4 && results && (
        <div style={{ maxWidth: 860, margin: "0 auto", padding: "32px 20px" }} className="slide-in">
          {/* Summary banner */}
          {results.overall_summary && (
            <div style={{ background: "linear-gradient(135deg, #0d1f3c, #0a1628)", border: "1px solid #1e3a5f", borderRadius: 12, padding: "18px 22px", marginBottom: 28, display: "flex", gap: 14, alignItems: "flex-start" }}>
              <span style={{ fontSize: 22, flexShrink: 0 }}>🧭</span>
              <div>
                <div className="syne" style={{ fontSize: 12, fontWeight: 700, color: "#3b82f6", textTransform: "uppercase", letterSpacing: ".08em", marginBottom: 6 }}>AI Summary</div>
                <p style={{ fontSize: 14, lineHeight: 1.65, color: "#cbd5e1" }}>{results.overall_summary}</p>
              </div>
            </div>
          )}

          <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 20, flexWrap: "wrap", gap: 10 }}>
            <div>
              <h2 className="syne" style={{ fontSize: 22, fontWeight: 800 }}>Priority Opportunities</h2>
              <p style={{ fontSize: 13, color: "#475569", marginTop: 2 }}>
                {results.opportunities?.length || 0} opportunities found · {results.non_opportunities?.length || 0} filtered out
              </p>
            </div>
            <button className="btn btn-secondary" onClick={() => { setStep(0); setResults(null); setChecked({}); setExpanded(null); }}>
              ↺ Start Over
            </button>
          </div>

          {/* Opportunity cards */}
          <div style={{ display: "flex", flexDirection: "column", gap: 16, marginBottom: 32 }}>
            {(results.opportunities || []).sort((a, b) => a.rank - b.rank).map((opp, i) => {
              const u = URGENCY[opp.urgency] || URGENCY.low;
              const tm = TYPE_META[opp.type] || TYPE_META.other;
              const isOpen = expanded === i;
              const totalActions = opp.action_checklist?.length || 0;
              const doneActions = opp.action_checklist?.filter((_, ai) => checked[`${i}-${ai}`]).length || 0;
              return (
                <div key={i} className="card" style={{ border: `1px solid ${isOpen ? "#1e3a5f" : "#1a2540"}`, transition: "border-color .2s" }}>
                  {/* Card header */}
                  <div style={{ padding: "18px 20px", cursor: "pointer" }} onClick={() => setExpanded(isOpen ? null : i)}>
                    <div style={{ display: "flex", alignItems: "flex-start", gap: 14 }}>
                      <div style={{ width: 36, height: 36, borderRadius: 8, background: "#1e2a4a", display: "flex", alignItems: "center", justifyContent: "center", fontSize: 18, flexShrink: 0 }}>{tm.icon}</div>
                      <div style={{ flex: 1, minWidth: 0 }}>
                        <div style={{ display: "flex", alignItems: "center", gap: 8, flexWrap: "wrap", marginBottom: 6 }}>
                          <span className="mono" style={{ fontSize: 11, color: "#475569" }}>#{opp.rank}</span>
                          <span className="badge" style={{ background: u.glow, color: u.color, border: `1px solid ${u.color}40`, ...(opp.urgency === "critical" ? { animation: "pulse 1.5s infinite" } : {}) }}>{u.label}</span>
                          <span className="badge" style={{ background: `${tm.color}15`, color: tm.color }}>{opp.type}</span>
                          {opp.stipend_or_benefit && <span className="badge" style={{ background: "#1e3a2a", color: "#2ed573" }}>💰 {opp.stipend_or_benefit}</span>}
                        </div>
                        <h3 className="syne" style={{ fontSize: 16, fontWeight: 700, marginBottom: 4, color: "#f1f5f9" }}>{opp.title}</h3>
                        <div style={{ fontSize: 13, color: "#64748b" }}>{opp.organization}</div>
                      </div>
                      <div style={{ flexShrink: 0, textAlign: "right" }}>
                        {opp.deadline && (
                          <div style={{ marginBottom: 8 }}>
                            <div className="mono" style={{ fontSize: 11, color: "#64748b", marginBottom: 2 }}>DEADLINE</div>
                            <div style={{ fontSize: 13, fontWeight: 600, color: u.color }}>{opp.deadline}</div>
                          </div>
                        )}
                        <div>
                          <div className="mono" style={{ fontSize: 11, color: "#64748b", marginBottom: 4 }}>PROFILE FIT</div>
                          <div style={{ fontSize: 18, fontWeight: 700, color: fitColor(opp.profile_fit_score), fontFamily: "'JetBrains Mono', monospace" }}>{opp.profile_fit_score}%</div>
                          <div className="fit-bar" style={{ width: 80, marginTop: 4 }}>
                            <div className="fit-fill" style={{ width: `${opp.profile_fit_score}%`, background: fitColor(opp.profile_fit_score) }} />
                          </div>
                        </div>
                      </div>
                      <div style={{ color: "#475569", fontSize: 18, flexShrink: 0, marginLeft: 4 }}>{isOpen ? "▲" : "▼"}</div>
                    </div>
                    {!isOpen && <p style={{ fontSize: 13, color: "#64748b", marginTop: 10, lineHeight: 1.5 }}>{opp.summary}</p>}
                  </div>

                  {/* Expanded details */}
                  {isOpen && (
                    <div style={{ borderTop: "1px solid #1a2540", padding: "20px 20px" }}>
                      <p style={{ fontSize: 14, color: "#94a3b8", lineHeight: 1.65, marginBottom: 20 }}>{opp.summary}</p>
                      <div className="grid2" style={{ marginBottom: 20 }}>
                        {/* Eligibility */}
                        <div>
                          <div className="mono" style={{ fontSize: 11, color: "#64748b", marginBottom: 10, letterSpacing: ".06em" }}>ELIGIBILITY REQUIREMENTS</div>
                          {opp.eligibility?.map((e, ei) => (
                            <div key={ei} style={{ display: "flex", gap: 8, marginBottom: 7, fontSize: 13, color: "#cbd5e1", alignItems: "flex-start" }}>
                              <span style={{ color: "#3b82f6", flexShrink: 0, marginTop: 2 }}>›</span> {e}
                            </div>
                          ))}
                        </div>
                        {/* Required docs */}
                        <div>
                          <div className="mono" style={{ fontSize: 11, color: "#64748b", marginBottom: 10, letterSpacing: ".06em" }}>REQUIRED DOCUMENTS</div>
                          {opp.required_documents?.length ? opp.required_documents.map((d, di) => (
                            <div key={di} style={{ display: "flex", gap: 8, marginBottom: 7, fontSize: 13, color: "#cbd5e1", alignItems: "flex-start" }}>
                              <span style={{ color: "#ffa502", flexShrink: 0, marginTop: 2 }}>📄</span> {d}
                            </div>
                          )) : <span style={{ fontSize: 13, color: "#475569" }}>None specified</span>}
                        </div>
                      </div>

                      {/* Why ranked + contact */}
                      <div className="grid2" style={{ marginBottom: 20 }}>
                        <div style={{ background: "#0d1226", borderRadius: 8, padding: "14px 16px" }}>
                          <div className="mono" style={{ fontSize: 11, color: "#64748b", marginBottom: 8, letterSpacing: ".06em" }}>WHY THIS RANK</div>
                          <p style={{ fontSize: 13, color: "#94a3b8", lineHeight: 1.6 }}>{opp.why_ranked_here}</p>
                        </div>
                        <div style={{ background: "#0d1226", borderRadius: 8, padding: "14px 16px" }}>
                          <div className="mono" style={{ fontSize: 11, color: "#64748b", marginBottom: 8, letterSpacing: ".06em" }}>PROFILE FIT REASONS</div>
                          {opp.profile_fit_reasons?.map((r, ri) => (
                            <div key={ri} style={{ fontSize: 13, color: "#94a3b8", marginBottom: 5, display: "flex", gap: 6 }}>
                              <span style={{ color: "#2ed573" }}>✓</span> {r}
                            </div>
                          ))}
                          {opp.profile_gaps?.map((g, gi) => (
                            <div key={gi} style={{ fontSize: 13, color: "#64748b", marginBottom: 5, display: "flex", gap: 6 }}>
                              <span style={{ color: "#ffa502" }}>△</span> {g}
                            </div>
                          ))}
                        </div>
                      </div>

                      {/* Action checklist */}
                      <div style={{ background: "#0a1428", border: "1px solid #1e3a5f", borderRadius: 10, padding: "16px 18px" }}>
                        <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 12 }}>
                          <div className="mono" style={{ fontSize: 11, color: "#3b82f6", letterSpacing: ".06em" }}>ACTION CHECKLIST</div>
                          <span style={{ fontSize: 12, color: doneActions === totalActions ? "#2ed573" : "#64748b" }}>{doneActions}/{totalActions} done</span>
                        </div>
                        {opp.action_checklist?.map((action, ai) => (
                          <label key={ai} className="check-item" style={{ opacity: checked[`${i}-${ai}`] ? 0.5 : 1 }}>
                            <input type="checkbox" checked={!!checked[`${i}-${ai}`]} onChange={() => toggleCheck(`${i}-${ai}`)} />
                            <span style={{ fontSize: 13, color: checked[`${i}-${ai}`] ? "#475569" : "#cbd5e1", textDecoration: checked[`${i}-${ai}`] ? "line-through" : "none", lineHeight: 1.5 }}>{action}</span>
                          </label>
                        ))}
                        {opp.contact_or_link && (
                          <div style={{ marginTop: 12, paddingTop: 12, borderTop: "1px solid #1a2540", display: "flex", alignItems: "center", gap: 8 }}>
                            <span style={{ fontSize: 12, color: "#64748b" }}>Apply/Contact:</span>
                            <span className="mono" style={{ fontSize: 12, color: "#3b82f6" }}>{opp.contact_or_link}</span>
                          </div>
                        )}
                      </div>
                    </div>
                  )}
                </div>
              );
            })}
          </div>

          {/* Non-opportunities */}
          {results.non_opportunities?.length > 0 && (
            <div>
              <div className="mono" style={{ fontSize: 11, color: "#475569", marginBottom: 12, letterSpacing: ".06em" }}>FILTERED OUT — NOT OPPORTUNITIES ({results.non_opportunities.length})</div>
              <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
                {results.non_opportunities.map((ne, i) => (
                  <div key={i} style={{ background: "#09091a", border: "1px solid #111827", borderRadius: 8, padding: "12px 16px", display: "flex", alignItems: "center", gap: 12 }}>
                    <span style={{ fontSize: 16 }}>🗑️</span>
                    <div>
                      <div style={{ fontSize: 13, color: "#475569", marginBottom: 2 }}>{ne.subject}</div>
                      <div style={{ fontSize: 12, color: "#374151" }}>{ne.reason}</div>
                    </div>
                  </div>
                ))}
              </div>
            </div>
          )}
        </div>
      )}
    </div>
  );
}
